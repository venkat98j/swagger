public static void main(String[] args) {
        // Create a ModuleFinder based on the current class loader
        ModuleFinder finder = ModuleFinder.of(ClassLoader.getSystemClassLoader());

        // Find all modules in the module path
        Set<ModuleReference> moduleReferences = finder.findAll();

        // Iterate over modules and print information
        for (ModuleReference moduleReference : moduleReferences) {
            String moduleName = moduleReference.descriptor().name();
            Set<String> opens = moduleReference.descriptor().opens();
            
            System.out.println("Module: " + moduleName);
            
            if (!opens.isEmpty()) {
                System.out.println("Opens:");
                for (String openedPackage : opens) {
                    System.out.println("  " + openedPackage);
                }
            } else {
                System.out.println("No opens directive for this module.");
            }

            System.out.println();
        }
    }
