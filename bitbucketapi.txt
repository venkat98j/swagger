import org.springframework.boot.SpringApplication;
import org.springframework.boot.SpringApplicationRunListener;
import org.springframework.core.Ordered;
import org.springframework.core.env.ConfigurableEnvironment;

public class CustomRunListener implements SpringApplicationRunListener, Ordered {

    public CustomRunListener(SpringApplication application, String[] args) {
        // Constructor
    }

    @Override
    public void starting() {
        // Executed when the application is starting
    }

    @Override
    public void environmentPrepared(ConfigurableEnvironment environment) {
        // Executed when the environment is prepared
    }

    @Override
    public void contextPrepared() {
        // Executed when the context is prepared
    }

    @Override
    public void contextLoaded() {
        // Executed when the context is loaded
    }

    @Override
    public void started() {
        // Executed when the application has started
    }

    @Override
    public void running() {
        // Executed when the application is running
    }

    @Override
    public void failed(ConfigurableEnvironment environment, Throwable exception) {
        // Executed when the application startup fails
        // Handle the exception appropriately
        exception.printStackTrace();
    }

    @Override
    public int getOrder() {
        // Define the order of the listener
        return Ordered.HIGHEST_PRECEDENCE;
    }
}
